def BinarySearch(n,list1,key):
  low = 0
  high = n-1
  while low <= high:
    mid = int(low + (high - low)/2)
  	if list1[mid] == key:
      return mid
    elif list1[mid] > key:
      high = mid - 1
    else:
      low = mid + 1
   return -1
